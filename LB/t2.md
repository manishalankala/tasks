~~~

kind: Deployment
apiVersion: apps/v1
metadata:
  name: frontend
  labels:
    app: frontend
spec:
  replicas: 1
  selector:
    matchLabels:
      app: frontend
  template:
    metadata:
      labels:
        app: frontend
    spec:
      containers:
        - name: linux
          image: docker pull jelastic/nodejs:16.8.0-npm
          imagePullPolicy: Always
          readinessProbe:
            failureThreshold: 2
            httpGet:
              path: /health
              port: 80
              scheme: HTTP
            initialDelaySeconds: 2
            periodSeconds: 2
            successThreshold: 1
            timeoutSeconds: 1
          livenessProbe:
            failureThreshold: 3
            httpGet:
              path: /health
              port: 80
              scheme: HTTP
            initialDelaySeconds: 5
            periodSeconds: 10
            successThreshold: 1
            timeoutSeconds: 1
          
          env:
          - name: "REACT_APP_AUTH_PASS"
            valueFrom:
              secretKeyRef:
                name: react-id
                key: REACT_APP_AUTH_PASS
          - name: "REACT_APP_AUTH_USER"
            valueFrom:
              secretKeyRef:
                name: react-id
                key: REACT_APP_AUTH_USER    
          ports: 
          - containerPort: 80
      imagePullSecrets:
      - name: dockerpullsecret


---         
          
apiVersion: v1
kind: Service
metadata:
  name: frontend-service
  labels:
    app: frontend-service
spec:
  ports:
  - port: 80
    targetPort: 80
  selector:
    app: frontend
